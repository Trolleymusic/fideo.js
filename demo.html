	// Call fideo as many times as you need to
	fideo({ "target": "#fideo-alaska", "size": "16:9" });

	// Leaving the size off defaults to 16:9
	fideo({ "target": "#fideo-alaska" });

	// You can also give it an HTML node instead of a selector
	var fideoAlaska = document.querySelectorAll("#fideo-alaska")[0];
	fideo({ "target": fideoAlaska, "size": "2.39:1" });
